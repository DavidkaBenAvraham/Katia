82 """Get estimated results.

98 def _get_organic(self) -> list:
99 """Get organic results.
100
101 This method returns the list of organic results. The data returned by this method doesn't
102 contain other search features like featured snippets, people also ask section etc.
103
104 Returns:
105 A list of organic Google Search results is returned.
106 """
107 organic = []
108 for g in self.tree.xpath('//div[@class="g"]'):
109 snippets = g.xpath('.//div/div/div[2]/div')
110 snippet = None
111 rich_snippet = None
112 if len(snippets) == 1:
113 snippet = snippets[0].text_content()
114 elif len(snippets) > 1:
115 if len(snippets[1].xpath('.//g-review-stars')) > 0:
116 rich_snippet = snippets[1].text_content()
117 snippet = snippets[0].text_content()
118 else:
119 snippet = snippets[1].text_content()
120 rich_snippet = snippets[0].text_content()
121
122 res = {
123 'url': self._clean(g.xpath('.//@href[1]')[0]),
124 'title': self._clean(g.xpath('.//h3/text()')[0]),
125 'snippet': self._clean(snippet),
126 'rich_snippet': self._clean(rich_snippet),
127 }
128 organic.append(res)
129 return organic

131 def _get_featured_snippet(self) -> dict:
132 """Get the featured snippet if exists.
133
134 Get the featured snippet (title, url) if it exists. If it doesn't exist,
135 then None is returned.
136
137 Returns:
138 any: A dict if featured snippet is found and None otherwise. i.e.::
139
140 {
141 'title': 'Example site title',
142 'url': 'http://example.com'
143 }
144
145 """
146 fs = None
147 snipp_el = self.tree.xpath(
148 '//div[contains(concat(" ", @class, " "), "kp-blk")]')
149 if len(snipp_el) > 0:
150 snipp_el = snipp_el[0]
151 heading = snipp_el.xpath('.//h3/text()')
152 url = snipp_el.xpath('.//a/@href')
153 if all([len(item) > 0 for item in [heading, url]]):
154 fs = {
155 'title': heading[0],
156 'url': url[-1]
157 }
158

161 def _get_knowledge_card(self) -> dict:
162 """Gets the knowledge card data if exists.
163
164 For some search queries, for example search queries against bigger brands, and celebs, Google
165 returns a knowledge card. It contains detailed information about the searched entity. Generally,
166 Google sources this data from Wikipedia as well as it uses its own database. If this card exists,
167 this method returns a dictionary, otherwise None.
168
169 Returns:
170 A dictionary if knowledge card exists, or None if it doesn't exist.
171 """
172 kc_el = self.tree.xpath('//div[contains(concat(" ", @class, " "), "kp-wholepage")]')
173 if len(kc_el):
174 kc_el = kc_el[0]
175 more_info = []
176 for el in kc_el.xpath('.//div[contains(@data-attrid, ":/")]'):
177 el_parts = el.xpath('.//span')
178 if len(el_parts) == 2:
179 more_info.append({
180 self._normalize_dict_key(el_parts[0].text_content()): el_parts[1].text_content()
181 })
182 else:
183 heading = el.xpath('.//div[@role="heading"]')
184 if len(heading) > 0:
185 heading_anchor = heading[0].xpath('.//a')
186 if len(heading_anchor) > 0:
187 dict_key = self._normalize_dict_key(heading_anchor[0].text_content())
188
189 dict_items = []
190 for item_div in el.xpath('.//div[@role="list"]'):
191
192 # Get list items
193 for item in item_div.xpath('.//div[@role="heading"]'):
194 if len(item):
195 dict_items.append({
196 'title': item.xpath('.//div[@class="title"]')[0].text_content(),
197 'subtitle': item.xpath('.//div')[1].text_content()
198 })
199
200 if dict_key == 'people_also_search_for':
201 for pasf in el.xpath('.//div[@data-reltype="sideways"]'):
202 dict_items.append(pasf.text_content())
203
204
205 more_info.append({
206 dict_key: dict_items
207 })
208
209 return {
210 'title': kc_el.xpath('.//h2/span')[0].text_content(),
211 'subtitle': kc_el.xpath('.//div[contains(@data-attrid, "subtitle")]')[0].text_content(),
212 'description': kc_el.xpath('.//div[@class="kno-rdesc"]/span')[0].text_content(),
213 'more_info': more_info
214 }
215
216 return None

218 def _get_scrolling_sections(self) -> list:
219 """Get data in scrolling widgets.

221 Sometimes, Google shows extra results, i.e. top stories and Tweets, in a scrolling widget. This method
222 will parse those results if available.
223
224 Returns:
225 list: Returns a list of the results. The list will either contain results or it will be
226 empty if no results are found.
227 """
228 sections = self.tree.xpath('//g-section-with-header')
229
230 data = []
231 if len(sections):
232 for section in sections:
233 section_title = section.xpath('.//h3')
234 if section_title:
235 title = section_title[0].text_content()
236 if title:
237 section_data = []
238 data_sections = section.xpath('.//g-inner-card')
239 if len(data_sections):
240 for data_section in data_sections:
241 data_title = data_section.xpath('.//div[@role="heading"]/text()')
242 data_url = data_section.xpath('.//a/@href')
243
244 if all(len(item) > 0 for item in [data_title, data_url]):
245 section_data.append({
246 'title': self._clean(data_title[0]),
247 'url': self._clean(data_url[0])
248 })
249 data.append({
250 'section_title': title,
251 'section_data': section_data
252 })

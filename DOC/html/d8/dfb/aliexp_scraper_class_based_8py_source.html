141 def extract_group_links(self):
142 '''
143 This function extracts group links from an AliExpress store.
144 Returns: Links to the subgroups.
145 '''
146
147 print(f'[PROCESS]: Executing Extraction function.')
148
149 pause_time = 2
150 last_height = self.driver.execute_script("return document.body.scrollHeight")
151
152 # Record the starting time
153 start = datetime.datetime.now()
154
155 print(f'[PROCESS]: Scrolling to the bottom of the page')
156 while True:
157 # Scroll down to bottom
158 self.driver.execute_script("window.scrollTo(0, document.body.scrollHeight);")
159
160 # wait to load page
161 time.sleep(pause_time)
162
163 # Calculate new scroll height and compare with last scroll height
164 new_height = self.driver.execute_script("return document.body.scrollHeight")
165 if new_height == last_height: # which means end of page
166 break
167 # update the last height
168 last_height = new_height
169
170 print(f'[INFO]: Scrolled to the bottom of the page.')
171
172 # Record the end time, then calculate and print the total time
173 end = datetime.datetime.now()
174 delta = end-start
175 print("[INFO] Total time taken to scroll till the end {}".format(delta))
176
177
178 # Extract sub-group-item class elements
179 print(f'[PROCESS]: Extracting link elements within sub-group-item class.')
180 sub_gp_element_list = self.driver.find_elements_by_css_selector(".sub-group-item a")
181 print(f'[INFO]: Extracted link elements successfully.')
182
183 # Extract all anchor tags
184 # link_tags = driver.find_elements_by_tag_name('a')
185
186 print(f'[PROCESS]: Extracting link text from link elements.')
187 # Create an emply list to hold all the urls for the sub groups
188 sub_gp_links = []
189
190 # Extract the urls of only the images from each of the tag WebElements
191 for tag in sub_gp_element_list:
192 sub_gp_links.append(tag.get_attribute('href'))
193 print(f'[INFO]: Extracted link text from link elements successfully.')
194
195 print(f'[PROCESS]: Printing Extracted links.')
196 for link in sub_gp_links:
197 print(link)
198 print(f'[INFO]: Print successful.')
199
200 return(sub_gp_links)

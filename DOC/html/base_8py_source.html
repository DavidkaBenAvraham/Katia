42def sign(secret, parameters):
43 #===========================================================================
44 # '''签名方法
45 # @param secret: 签名需要的密钥
46 # @param parameters: 支持字典和string两种
47 # '''
48 #===========================================================================
49 # 如果parameters 是字典类的话
50 if hasattr(parameters, "items"):
51 keys = parameters.keys()
52 keys = list(keys)
53 keys.sort()
54
55 parameters = "%s%s%s" % (secret,
56 str().join('%s%s' % (key, parameters[key]) for key in keys),
57 secret)
58 parameters = parameters.encode('utf-8')
59 sign = hashlib.md5(parameters).hexdigest().upper()
60 return sign

100 def __str__(self):
101 """Return a string representing the form data, including attached files."""
102 # Build a list of lists, each containing "lines" of the
103 # request. Each part is separated by a boundary string.
104 # Once the list is built, return a string where each
105 # line is separated by '\r\n'.
106 parts = []
107 part_boundary = '--' + self.boundary
108
109 # Add the form fields
110 parts.extend(
111 [ part_boundary,
112 'Content-Disposition: form-data; name="%s"' % name,
113 'Content-Type: text/plain; charset=UTF-8',
114 '',
115 value,
116 ]
117 for name, value in self.form_fields
118 )
119
120 # Add the files to upload
121 parts.extend(
122 [ part_boundary,
123 'Content-Disposition: file; name="%s"; filename="%s"' % \
124 (field_name, filename),
125 'Content-Type: %s' % content_type,
126 'Content-Transfer-Encoding: binary',
127 '',
128 body,
129 ]
130 for field_name, filename, content_type, body in self.files
131 )
132
133 # Flatten the list and add closing boundary marker,
134 # then return CR+LF separated data
135 flattened = list(itertools.chain(*parts))
136 flattened.append('--' + self.boundary + '--')
137 flattened.append('')
138 return '\r\n'.join(flattened)
